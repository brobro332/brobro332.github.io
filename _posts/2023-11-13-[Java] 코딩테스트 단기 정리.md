### 2️⃣ 탐색

- 예제1: 기존에 파이썬으로 풀었던 문제를 자바로 옮겨보았다.

```java
import java.io.*;
import java.util.*;

public class Main {

    public static void dfs(int y, int x, int[][] visited, char[][] graph) {

        visited[y][x] = 1;
        int[] direction = {-1, 1};

        if (graph[y][x] == '-') {

            for (int i = 0; i < direction.length; i++) {
                int nx = x + direction[i];

                if (nx >= 0 && nx < graph[0].length) {
                    if (graph[y][nx] == '-' && visited[y][nx] == 0) {
                        visited[y][nx] = 1;
                        dfs(y, nx, visited, graph);
                    }
                }
            }
        }
        else {

            for (int i = 0; i < direction.length; i++) {
                int ny = y + direction[i];

                if (ny >= 0 && ny < graph.length) {
                    if (graph[ny][x] == '|' && visited[ny][x] == 0) {
                        visited[ny][x] = 1;
                        dfs(ny, x, visited, graph);
                    }
                }
            }
        }
    }

    public static void main(String[] args) throws Exception {

        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        StringTokenizer st = new StringTokenizer(br.readLine());

        int N = Integer.parseInt(st.nextToken());
        int M = Integer.parseInt(st.nextToken());

        char[][] graph = new char[N][M];

        for (int i = 0; i < N; i++) {
            String temp = br.readLine();
            for (int j = 0; j < M; j++) {
                graph[i][j] = temp.charAt(j);
            }
        }

        int[][] visited = new int[N][M];
        for (int i = 0; i < N; i++) {
            for (int j = 0; j < M; j++) {
                visited[i][j] = 0;
            }
        }

        int cnt = 0;
        for (int i = 0; i < N; i++) {
            for (int j = 0; j < M; j++) {
                if (visited[i][j] == 0) {
                    dfs(i, j, visited, graph);
                    cnt += 1;
                }
            }
        }

        System.out.println(cnt);
    }
}
```

- 예제2: System.out.print()를 너무 많이 시간복잡도 대폭 상승하므로 StringBuilder 사용해야함

```java
import java.io.*;
import java.util.*;
public class Main {
    static int N;
    static int M;
    static StringBuilder sb = new StringBuilder();

    public static void dfs(int[] arr, List<Integer> result, int depth) {
        if (depth == M) {
            for (int i = 0; i < result.size(); i++) {
                sb.append(result.get(i) + " ");
            }
            sb.append("\n");

            return;
        }

        for (int i = 0; i < arr.length; i++) {
            result.add(arr[i]);

            dfs(arr, result, depth + 1);

            result.remove(result.size() - 1);
        }
    }

    public static void main(String[] args) throws Exception {

        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        StringTokenizer st1 = new StringTokenizer(br.readLine());
        StringTokenizer st2 = new StringTokenizer(br.readLine());

        N = Integer.parseInt(st1.nextToken());
        M = Integer.parseInt(st1.nextToken());

        List<Integer> result = new ArrayList<>();

        int[] nums = new int[N];
        for (int i = 0; i < N; i++) {
            nums[i] = Integer.parseInt(st2.nextToken());
        }

        Arrays.sort(nums);

        dfs(nums, result, 0);

        System.out.println(sb);
    }
}
```
